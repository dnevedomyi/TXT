На курсе GiT я научусь следующим скилам:
git init # Инициализация/создание репозитория
git help # справка по всем командам
git clone #пененос данных с внешного репозитория
git status #узнать статус документа 
git commit -m "<комментарий>" #установить комментарий
git branch # просмотреть все ветви, отслеживая его версии
git add . #добавить все файлы проекта в нам будующий commit
git add <имя_файла>#добавить конкретный файл на будующий commit
git checkout #отменяет изменения файла
git merge #используется для слияния одной или нескольких веток в текущую. Затем она устанавливает указатель текущей ветки на результирующий коммит.
git remote #просмотр истории удалённых репозиториев
git fetch #получение изменений с сервера.Git собирает все коммиты из целевой ветки, которых нет в текущей ветке, и сохраняет их в локальном репозитории.
git push #отправить данные на внешний репозиторий
git pull #получить измененные данные файлов из внешнего репозитория.Он сливает любые внесённые коммиты в ветку, в которой вы сейчас работаете
